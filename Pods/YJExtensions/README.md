# YJExtensions
常用系统类扩展

## 使用方式

1、集成:

```
pod 'YJExtensions'
```

2、扩展类

```
/** UIKit */
#import "UIImage+YJ.h"
#import "UIView+YJ.h"
#import "UIWindow+YJ.h"
#import "UIButton+YJ.h"
#import "UIDevice+YJ.h"
#import "UIViewController+YJ.h"
#import "UINavigationBar+YJ.h"
#import "UIImageView+YJ.h"

/** Foundation */
#import "NSArray+YJ.h"
#import "NSDictionary+YJ.h"
#import "NSObject+YJ.h"
#import "NSBundle+YJ.h"
#import "NSString+YJ.h"
#import "NSDate+YJ.h"
#import "NSNumber+YJ.h"
#import "NSUserDefaults+YJ.h"
#import "NSMutableAttributedString+YJ.h"
#import "NSFileManager+YJ.h"
```
